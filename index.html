<script>

function createWave(container) {

  return WaveSurfer.create({

    container: container,

    waveColor: '#5F5F5F',

    progressColor: '#F0F0F0',

    backend: 'MediaElement',

    height: 140,

  });

}


const before = createWave('#waveBefore');

const after = createWave('#waveAfter');


before.load(‘SampleBefore.wav');

after.load(‘SampleAfterwav');


let playing = false;

let isSyncing = false;


const playBtn = document.getElementById('playBtn');

const beforeBtn = document.getElementById('beforeBtn');

const afterBtn = document.getElementById('afterBtn');



// =======================

// LOCK SEEK SYNC (SAFE)

// =======================

function lockSync(source, target) {

  source.on('seeking', () => {

    if (isSyncing) return;


    isSyncing = true;

    const time = source.getCurrentTime();

    target.setTime(time);

    isSyncing = false;

  });

}


lockSync(before, after);

lockSync(after, before);



// =======================

// RELIABLE LOOPING

// =======================

function enableLoop(ws1, ws2) {

  ws1.on('finish', () => {

    if (!playing) return;


    ws1.setTime(0);

    ws2.setTime(0);


    ws1.play();

    ws2.play();

  });

}


enableLoop(before, after);



// =======================

// PLAY / PAUSE

// =======================

playBtn.onclick = () => {

  if (!playing) {

    before.play();

    after.play();

    playing = true;

  } else {

    before.pause();

    after.pause();

    playing = false;

  }

};



// =======================

// SPACEBAR TOGGLE

// =======================

document.addEventListener('keydown', e => {

  if (e.code === 'Space') {

    e.preventDefault();

    playBtn.click();

  }

});



// =======================

// A/B SWITCH

// =======================

function switchTrack(track) {

  if (track === 'before') {

    before.setVolume(1);

    after.setVolume(0);

    beforeBtn.classList.add('active');

    afterBtn.classList.remove('active');

  } else {

    before.setVolume(0);

    after.setVolume(1);

    afterBtn.classList.add('active');

    beforeBtn.classList.remove('active');

  }

}


beforeBtn.onclick = () => switchTrack('before');

afterBtn.onclick = () => switchTrack('after');



// Default state

switchTrack('before');

</script>
